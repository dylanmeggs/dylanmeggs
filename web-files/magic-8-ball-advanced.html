import sys
import random
import csv

class magic8ball:
    def __init__(self, name):
        self.__name = name #properties of init method
        self.__mQuestions = []
        self.__start_game() #initializing a class calls the start game function

    def __start_game(self):
        responses = ["IT IS CERTAIN", "YOU MAY RELY ON IT", "AS I SEE IT, YES", "OUTLOOK LOOKS GOOD", "MOST LIKELY", "IT IS DECIDELY SO", "WITHOUT A DOUBT", "YES, DEFINETLY"]
        questions = True
        print("Welcome " + self.__name)

        while questions:
            ques = input("Please ask a question (press enter to exit): ")
            if ques == "":
                print("Game ended.")
                self.__write_questions()
                sys.exit()
            else:
                print(responses[random.randint(0,7)])
                self.__mQuestions.append(ques)
# self.__write_questions() adding this here causes duplications. It prints the whole list, so each time you ask a new question, it also writes the old ones again.
# exiting the program (via enter or any other method) causes it to print the whole list again since I have it included in the first if statement

    def __write_questions(self):
        file = open("magic_questions.csv", "a", newline="")
        wrt = csv.writer(file) # wrt is new variable = the csv writer function
        
        for x in self.__mQuestions:
            wrt.writerow([x])
        file.close()

-- Table for questions
CREATE TABLE analytics.magic8_questions (
    question_id INT IDENTITY(1,1),
    question_text VARCHAR(255)
);

INSERT INTO analytics.magic8_questions (question_text)
VALUES
('Should I hire Dylan?'),
('Is Dylan the right candidate for my Engineering role?'),
('Will Dylan excel in this position?');

-- Table for answers
CREATE TABLE analytics.magic8_answers (
    answer_id INT IDENTITY(1,1),
    question_id INT REFERENCES analytics.magic8_questions(question_id),
    answer_text VARCHAR(255)
);

INSERT INTO analytics.magic8_answers (question_id, answer_text)
SELECT q.question_id, a.answer_text
FROM analytics.magic8_questions q
CROSS JOIN (
    VALUES ('Yes'), ('Absolutely'), ('Without a Doubt'), ('Definitely')
) AS a(answer_text);

import boto3
import random
import time

# Redshift Data API client (IAM role-based)
client = boto3.client("redshift-data")

cluster_id = "my-redshift-cluster"
database = "analytics"
db_user = "awsuser"

# Query questions from Redshift
query_questions = """
SELECT question_id, question_text
FROM analytics.magic8_questions
ORDER BY question_id;
"""

response = client.execute_statement(
    ClusterIdentifier=cluster_id,
    Database=database,
    DbUser=db_user,
    Sql=query_questions
)

query_id = response["Id"]
while True:
    status = client.describe_statement(Id=query_id)
    if status["Status"] in ["FINISHED", "FAILED", "ABORTED"]:
        break
    time.sleep(1)

result = client.get_statement_result(Id=query_id)
questions = [
    (row[0].get("longValue"), row[1].get("stringValue"))
    for row in result["Records"]
]
